Resource Share

Provides a RAM Resource Share

exports.createResources = () => [
  {
    type: "ResourceShare",
    group: "RAM",
    properties: ({}) => ({
      allowExternalPrincipals: false,
      featureSet: "STANDARD",
      name: "ipam-org-share",
    }),
  },
];
