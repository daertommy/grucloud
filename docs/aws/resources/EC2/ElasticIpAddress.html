Elastic Ip Address

Provides an Elastic Ip Address to be associated to an EC2 instance

exports.createResources = () => [
  {
    type: "ElasticIpAddress",
    group: "EC2",
    name: "eip",
    properties: ({}) => ({
      Tags: [
        {
          Key: "mykey",
          Value: "myvalue",
        },
      ],
    }),
  },
];
