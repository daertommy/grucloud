User Pool

Manages a Cognito User Pool.

Sample code​

exports.createResources = () => [
  {
    type: "UserPool",
    group: "CognitoIdentityServiceProvider",
    name: "my-user-pool",
    properties: () => ({
      UserPoolTags: {
        mykey1: "myvalue",
      },
    }),
  },
];
